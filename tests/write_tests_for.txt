root_dir = r'tests\data'

cif_path = r''

sets = list(amd.CifReader(cif_path))

f = open(os.path.join(root_dir, rf'{set_name}.pkl'), 'wb')
sets = []
for s in amd.CifReader(cif_path):
    s.tags['amd'] = amd.AMD(s, 100)
    s.tags['pdd'] = amd.PDD(s, 100)
    sets.append(s)

with amd.SetWriter(os.path.join(root_dir, rf'{set_name}.hdf5')) as w:
    w.iwrite(sets)

with open(os.path.join(root_dir, rf'{set_name}.pkl'), 'wb') as f:
    pickle.dump(sets, f)
